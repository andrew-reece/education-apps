** TABLE NAME: user_access **

id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT,
PRIMARY KEY(id),
username VARCHAR(15),
password TEXT,
access_level TINYINT(1) UNSIGNED,
student_id SMALLINT UNSIGNED NOT NULL,
firstname VARCHAR(15),
surname VARCHAR(15),
email VARCHAR(40)

** TABLE NAME: activity_log **

id MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,
PRIMARY KEY(id),
timestamp DATETIME,
user VARCHAR(12),
action VARCHAR(12),
msg VARCHAR(255),
urgency VARCHAR(15)
